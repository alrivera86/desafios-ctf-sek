// Jenkinsfile — deploy-service (PelicanCorp) // GitHub org: https://github.com/tinmarino/pelicancorp/ pipeline { agent any parameters { string(name: 'SERVICE_NAME', defaultValue: 'webapp', description: 'Microservice name to deploy') } environment { GITHUB_TOKEN = credentials('github-deploy-key') } stages { stage('Install') { steps { sh ''' git clone https://github.com/pelicancorp/$SERVICE_NAME.git cd $SERVICE_NAME ''' } } stage('Pre-Build') { steps { sh ''' cd $SERVICE_NAME sh test.sh ''' } } stage('Build') { steps { sh ''' cd $SERVICE_NAME ./publish.sh ''' } } } post { always { cleanWs() } } }
